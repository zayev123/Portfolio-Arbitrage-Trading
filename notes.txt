SELECT 
    SUBQ.sub_pairs_list,
    P.symbol,
    P.ID
FROM 
    parent_pairs P
JOIN (
    SELECT 
        parent_pairs.ID AS parent_id,
        STRING_AGG(child_pairs.ID::TEXT, ',') AS sub_pairs_list
    FROM 
        parent_pairs
    LEFT JOIN 
        child_pairs ON child_pairs.parent_pair_id = parent_pairs.ID
    GROUP BY 
        parent_pairs.ID
) AS SUBQ ON SUBQ.parent_pair_id = P.ID;
